lab5
========
 docker run -it --name dev1 ubuntu /bin/bash
 apt-get update
 apt-get install openjdk-8-jdk
 apt-get install maven
 apt-get install git
 
 Dockerfile:
 FROM ubuntu
 EXPOSE 8080
 RUN apt-get update && apt-get install -y \
  openjdk-8-jdk \
  maven \
  git 
  ENTRYPOINT ["/bin/bash"]
 
 docker built -t dev_img .
